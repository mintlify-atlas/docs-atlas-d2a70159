Product Management is the core inventory system in Bodeguita, providing complete CRUD operations, category management, stock tracking, and advanced search capabilities.

## Overview

The product module consists of two main screens:
- **ProductListScreen**: Browse, search, filter, and manage existing products
- **ProductRegistrationScreen**: Create new products or edit existing ones

### Data Model

Products are stored using Room with the following structure:

```kotlin
// Product.kt
@Entity(
    tableName = "producto",
    indices = [
        Index(value = ["code"], unique = true),
        Index(value = ["category_id"])
    ],
    foreignKeys = [
        ForeignKey(
            entity = Category::class,
            parentColumns = ["id"],
            childColumns = ["category_id"],
            onDelete = ForeignKey.SET_NULL
        )
    ]
)
data class Product(
    @PrimaryKey(autoGenerate = true) val id: Int = 0,
    val name: String,
    val code: String,        // Unique identifier
    val price: Double,
    val stock: Int,
    val descripcion: String,
    @ColumnInfo(name = "category_id") val categoryId: Long? = null
)
```

<Note>
  Product codes are **unique** and case-insensitive. The system automatically converts codes to uppercase.
</Note>

Implements efficient client-side filtering across multiple fields:

```kotlin
// ProductListScreen.kt:48
var query by rememberSaveable { mutableStateOf("") }

// Reactive filtering
val filteredProducts by remember(products, query) {
    derivedStateOf {
        val q = query.trim().lowercase()
        if (q.isBlank()) products
        else products.filter { p ->
            val name = p.name.lowercase()
            val code = p.code.lowercase()
            val desc = p.descripcion?.lowercase().orEmpty()
            name.contains(q) || code.contains(q) || desc.contains(q)
        }
    }
}
```

```kotlin
@Composable
private fun StockBadge(stock: Int) {
    val (bg, fg) = when {
        stock == 0 -> ErrorColor to Color.White      // Red for no stock
        stock < 10 -> WarningColor to Color.White    // Yellow for low stock
        else -> PrimaryGreen to Color.White          // Green for available
    }
    Surface(
        color = bg,
        shape = MaterialTheme.shapes.small
    ) {
        Text(
            text = when {
                stock == 0 -> "Sin Stock"
                stock < 10 -> "Stock Bajo"
                else -> "Disponible"
            },
            modifier = Modifier.padding(horizontal = 8.dp, vertical = 4.dp),
            style = MaterialTheme.typography.bodySmall,
            color = fg
        )
    }
}
```

## Save Operation

Upsert logic with category resolution:

```kotlin
// ProductRegistrationScreen.kt:244
PrimaryButton(
    text = "Guardar Producto",
    onClick = {
        val ok = nombre.isNotBlank() && codigo.isNotBlank() && precio != null && stock != null
        if (!ok) {
            scope.launch { snackbar.showSnackbar("Completa todos los campos con valores válidos") }
            return@PrimaryButton
        }

        scope.launch {
            try {
                // Resolve categoryId
                var catId: Long? = null
                val texto = categoriaNombre.trim()
                if (texto.isNotEmpty()) {
                    val existente = categories.firstOrNull {
                        it.nombre.equals(texto, ignoreCase = true)
                    }
                    if (existente != null) {
                        catId = existente.id
                    } else {
                        // Create and get id
                        val newId = categoriaDao.upsertByName(texto)
                        catId = if (newId != -1L) newId else categories
                            .firstOrNull { it.nombre.equals(texto, ignoreCase = true) }?.id
                    }
                }

                val p = Product(
                    name = nombre.trim(),
                    code = codigo.trim().uppercase(),
                    price = precio!!,
                    stock = stock!!,
                    descripcion = descripcion.trim(),
                    categoryId = catId
                )

                // Upsert by code (update if exists, insert if new)
                productoDao.upsertByCode(p)

                snackbar.showSnackbar("Producto guardado")
                navController.navigateUp()
            } catch (e: Exception) {
                snackbar.showSnackbar("Error al guardar: ${e.message ?: "desconocido"}")
            }
        }
    },
    modifier = Modifier.weight(1f)
)
```

<Warning>
  Product codes are **unique**. If you save a product with an existing code, it will **update** the existing product rather than creating a duplicate.
</Warning>

## ProductViewModel

The ViewModel manages reactive product state:

```kotlin
// ProductViewModel.kt
class ProductViewModel(private val repository: ProductRepository) : ViewModel() {

    // Reactive product list from Room
    val products: StateFlow<List<Product>> =
        repository.observeAllProducts()
            .stateIn(
                scope = viewModelScope,
                started = SharingStarted.WhileSubscribed(5_000),
                initialValue = emptyList()
            )

    // Event channel for UI feedback
    private val _events = MutableSharedFlow<String>(extraBufferCapacity = 1)
    val events: SharedFlow<String> = _events.asSharedFlow()

    fun upsertProduct(product: Product) = viewModelScope.launch {
        try {
            repository.upsertByCode(product)
            _events.tryEmit("Producto guardado")
        } catch (e: Exception) {
            _events.tryEmit("Error al guardar: ${e.message ?: "desconocido"}")
        }
    }

    fun deleteProduct(product: Product) = viewModelScope.launch {
        repository.deleteProduct(product)
    }

    fun updateProduct(product: Product) = viewModelScope.launch {
        repository.updateProduct(product)
    }
}
```

## Best Practices

<Info>
  **Search Performance**: The search uses `derivedStateOf` to avoid unnecessary recompositions. Filtering happens in-memory on the client side, which is fast for typical store inventories (< 1000 products).
</Info>

<Note>
  **Form Validation**: 
  - Price and stock inputs show error states when non-numeric
  - Save button validates all required fields before submission
  - Decimal prices support both comma and period separators
</Note>

<Warning>
  **Category Foreign Key**: When a category is deleted, products linked to it have their `categoryId` set to `NULL` (not deleted) due to `onDelete = ForeignKey.SET_NULL`.
</Warning>
