## Overview

All screens are built with **Jetpack Compose** following Material Design 3 guidelines. Each screen is a `@Composable` function that integrates with ViewModels for state management and navigation.

## Common Pattern

All screens follow this structure:

```kotlin
@Composable
fun ScreenName(
    navController: NavController,
    viewModel: SomeViewModel,
    onMenuClick: () -> Unit
) {
    PageLayout(
        title = "Screen Title",
        onMenuClick = onMenuClick
    ) {
        // Screen content
    }
}
```

## ProductListScreen

Product catalog with search, filtering, and CRUD operations.

### Signature

```kotlin
@Composable
fun ProductListScreen(
    navController: NavController,
    productViewModel: ProductViewModel,
    onMenuClick: () -> Unit = {}
)
```

### Key Features

**1. Search Functionality**
```kotlin
var query by rememberSaveable { mutableStateOf("") }

val filteredProducts by remember(products, query) {
    derivedStateOf {
        val q = query.trim().lowercase()
        if (q.isBlank()) products
        else products.filter { p ->
            val name = p.name.lowercase()
            val code = p.code.lowercase()
            val desc = p.descripcion?.lowercase().orEmpty()
            name.contains(q) || code.contains(q) || desc.contains(q)
        }
    }
}

SearchField(
    value = query,
    onValueChange = { query = it },
    onClear = { query = "" },
    placeholder = "Buscar por nombre, código o descripción"
)
```

**2. Product Cards**
```kotlin
LazyColumn(verticalArrangement = Arrangement.spacedBy(8.dp)) {
    items(filteredProducts) { product ->
        ProductCard(
            product = product,
            onDelete = { productViewModel.deleteProduct(product) },
            onEdit = { navController.navigate("product_registration/${product.id}") }
        )
    }
}
```

**3. Stock Badge**
```kotlin
@Composable
private fun StockBadge(stock: Int) {
    val (bg, fg) = when {
        stock == 0 -> ErrorColor to Color.White
        stock < 10 -> WarningColor to Color.White
        else -> PrimaryGreen to Color.White
    }
    Surface(color = bg, shape = MaterialTheme.shapes.small) {
        Text(
            text = when {
                stock == 0 -> "Sin Stock"
                stock < 10 -> "Stock Bajo"
                else -> "Disponible"
            },
            style = MaterialTheme.typography.bodySmall,
            color = fg
        )
    }
}
```

### User Interactions

- Search products by name, code, or description
- Edit product details
- Delete products
- Navigate to product registration
- View stock status with color-coded badges

### Key Features

**1. Supplier Selection**
```kotlin
val suppliers by purchasesVM.suppliers.collectAsState(initial = emptyList())
var selectedSupplierIdx by remember { mutableStateOf(-1) }
val selectedSupplier = suppliers.getOrNull(selectedSupplierIdx)

ExposedDropdownMenuBox(
    expanded = expanded,
    onExpandedChange = { expanded = !expanded }
) {
    OutlinedTextField(
        value = selectedSupplier?.nombre ?: "",
        onValueChange = {},
        readOnly = true,
        label = { Text("Selecciona un proveedor") }
    )

    ExposedDropdownMenu(
        expanded = expanded,
        onDismissRequest = { expanded = false }
    ) {
        suppliers.forEachIndexed { index, s ->
            DropdownMenuItem(
                text = { Text("${s.nombre}  —  ${s.ruc}") },
                onClick = {
                    selectedSupplierIdx = index
                    expanded = false
                }
            )
        }
    }
}
```

**2. Draft Cart (Create Products On-The-Fly)**
```kotlin
Button(
    onClick = {
        val q = qtyTxt.toIntOrNull() ?: 0
        val c = costTxt.replace(',', '.').toDoubleOrNull() ?: -1.0
        
        when {
            nameOrCode.isBlank() ->
                scope.launch { snackbar.showSnackbar("Ingresa el nombre o código") }
            q <= 0 ->
                scope.launch { snackbar.showSnackbar("Cantidad inválida") }
            c < 0.0 ->
                scope.launch { snackbar.showSnackbar("Costo inválido") }
            else -> {
                purchasesVM.addDraftLine(nameOrCode, q, c)
                nameOrCode = ""
                qtyTxt = "1"
                costTxt = ""
            }
        }
    }
) {
    Icon(Icons.Default.Add, null)
    Text("Agregar")
}
```

**3. Finalize Purchase**
```kotlin
Button(
    onClick = {
        val supplier = selectedSupplier
        when {
            supplier == null ->
                scope.launch { snackbar.showSnackbar("Selecciona un proveedor") }
            cart.isEmpty() ->
                scope.launch { snackbar.showSnackbar("Agrega al menos un producto") }
            else -> {
                purchasesVM.finalizeDraftPurchase(supplier.id)
                navController.popBackStack()
            }
        }
    },
    enabled = cart.isNotEmpty()
) {
    Icon(Icons.Default.Save, null)
    Text("Registrar compra")
}
```

**4. Supplier Manager Dialog**
```kotlin
if (showSupplierManager) {
    SupplierManagerDialog(
        onDismiss = { showSupplierManager = false },
        onSave = { ruc, nombre, telefono ->
            purchasesVM.upsertSupplierByRuc(ruc, nombre, telefono)
        }
    )
}
```

## Navigation Integration

All screens integrate with Jetpack Navigation Compose:

```kotlin
NavHost(navController, startDestination = "login") {
    composable("login") {
        LoginScreen(navController)
    }
    composable("dashboard") {
        DashboardScreen(navController, productVM, salesVM, dashboardVM, onMenuClick)
    }
    composable("products") {
        ProductListScreen(navController, productVM, onMenuClick)
    }
    composable("product_registration/{productId}") { backStackEntry ->
        val productId = backStackEntry.arguments?.getString("productId")?.toIntOrNull()
        val product = products.find { it.id == productId }
        ProductRegistrationScreen(navController, productVM, categoryVM, product, onMenuClick)
    }
    composable("sales") {
        SalesScreen(navController, salesVM, onMenuClick)
    }
    composable("purchases") {
        PurchasesScreen(navController, purchasesVM, onMenuClick)
    }
    composable("cash_closure") {
        CashClosureScreen(navController, cashClosureVM, onMenuClick)
    }
    composable("profile") {
        ProfileScreen(navController, onMenuClick, themeManager)
    }
}
```
